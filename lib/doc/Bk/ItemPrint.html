def print_n2(level, name, dts , dls)
  puts (" "*level) + name if name

  if dts
    dts.each do |x| 
      x.print_n2(level + 1, x.name, x.dts, x.dls )
    end
  end
  if dls
    dls.each do |x| 
      x.print_n2(level + 1 , x.name , x.dts , x.dls )
    end
  end
end
